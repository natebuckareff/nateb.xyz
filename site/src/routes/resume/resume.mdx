export const SkillList = ({ title, items }) => {
    const list = intersperse(items, ', ').map(x => <i>{x}</i>)
    return (
        <li class="my-1">
            <span class="font-bold">{title}</span>: {list}
        </li>
    )
}

export const intersperse = (array, sep) => {
    const items = []
    for (let i = 0; i < array.length; ++i) {
        items.push(array[i])
        if (i < array.length - 1) {
            items.push(sep)
        }
    }
    return items
}
